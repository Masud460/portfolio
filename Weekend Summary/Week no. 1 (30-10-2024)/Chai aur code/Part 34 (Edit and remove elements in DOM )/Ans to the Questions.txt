1. Here it is:
    function addLanguage(langName) {
      const ul = document.querySelector(".languages");
      const li = document.createElement("li");
      li.innerHTML = `langName`;
      ul.appendChild(li);
    }
    addLanguage("Python");

2. Here it is:
    function addLanguage(langName) {
      const ul = document.querySelector(".languages");
      const li = document.createElement("li");
      const nameOfLang = document.createTextNode(langName);
      li.appendChild(nameOfLang);
      ul.appendChild(li);
    }
    addLanguage("Python");
